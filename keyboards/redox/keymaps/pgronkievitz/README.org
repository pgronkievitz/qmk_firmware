#+name: tappingterm
#+begin_src c
uint16_t get_tapping_term(uint16_t keycode, keyrecord_t *record) {
    switch (keycode) {
        case KC_CTES:
        case KC_CTQU:
            return 220;
        case KC_NAMI:
        case KC_NAGR:
            return 180;
        default:
            return TAPPING_TERM;
    }
}
